171 static Vector3 MoveTowards(Vector3 current, Vector3 target, float maxDistanceDelta) {
172 Vector3 a = target - current;
173 float magnitude = a.x * a.x + a.y * a.y + a.z * a.z;
174 if (magnitude <= maxDistanceDelta || magnitude == 0.0f) {
175 return target;
176 }
177 float num = sqrt(magnitude);
178 return current + a / num * maxDistanceDelta;
179 }
