27struct Sphere : public BoundingVolume
28{
29 glm::vec3 center{ 0.f, 0.f, 0.f };
30 float radius{ 0.f };
31
32 Sphere(const glm::vec3& inCenter, float inRadius)
33 : BoundingVolume{}, center{ inCenter }, radius{ inRadius }
34 {}
35
36 bool isOnOrForwardPlane(const Plane& plane) const final
37 {
38 return plane.getSignedDistanceToPlane(center) > -radius;
39 }
40
41 bool isOnFrustum(const Frustum& camFrustum, const Transform& transform) const final
42 {
43 //Get global scale thanks to our transform
44 const glm::vec3 globalScale = transform.getGlobalScale();
45
46 //Get our global center with process it with the global model matrix of our transform
47 const glm::vec3 globalCenter{ transform.getModelMatrix() * glm::vec4(center, 1.f) };
48
49 //To wrap correctly our shape, we need the maximum scale scalar.
50 const float maxScale = std::max(std::max(globalScale.x, globalScale.y), globalScale.z);
51
52 //Max scale is assuming for the diameter. So, we need the half to apply it to our radius
53 Sphere globalSphere(globalCenter, radius * (maxScale * 0.5f));
54
55 //Check Firstly the result that have the most chance to failure to avoid to call all functions.
56 return (globalSphere.isOnOrForwardPlane(camFrustum.leftFace) &&
57 globalSphere.isOnOrForwardPlane(camFrustum.rightFace) &&
58 globalSphere.isOnOrForwardPlane(camFrustum.farFace) &&
59 globalSphere.isOnOrForwardPlane(camFrustum.nearFace) &&
60 globalSphere.isOnOrForwardPlane(camFrustum.topFace) &&
61 globalSphere.isOnOrForwardPlane(camFrustum.bottomFace));
62 };
63};

65struct SquareAABB : public BoundingVolume
66{
67 glm::vec3 center{ 0.f, 0.f, 0.f };
68 float extent{ 0.f };

70 SquareAABB(const glm::vec3& inCenter, float inExtent)

74 bool isOnOrForwardPlane(const Plane& plane) const final
75 {
76 // Compute the projection interval radius of b onto L(t) = b.c + t * p.n
77 const float r = extent * (std::abs(plane.normal.x) + std::abs(plane.normal.y) + std::abs(plane.normal.z));
78 return -r <= plane.getSignedDistanceToPlane(center);
79 }

81 bool isOnFrustum(const Frustum& camFrustum, const Transform& transform) const final
82 {
83 //Get global scale thanks to our transform
84 const glm::vec3 globalCenter{ transform.getModelMatrix() * glm::vec4(center, 1.f) };
85
86 // Scaled orientation
87 const glm::vec3 right = transform.getRight() * extent;
88 const glm::vec3 up = transform.getUp() * extent;
89 const glm::vec3 forward = transform.getForward() * extent;
90
91 const float newIi = std::abs(glm::dot(glm::vec3{ 1.f, 0.f, 0.f }, right)) +
92 std::abs(glm::dot(glm::vec3{ 1.f, 0.f, 0.f }, up)) +
93 std::abs(glm::dot(glm::vec3{ 1.f, 0.f, 0.f }, forward));
94
95 const float newIj = std::abs(glm::dot(glm::vec3{ 0.f, 1.f, 0.f }, right)) +
96 std::abs(glm::dot(glm::vec3{ 0.f, 1.f, 0.f }, up)) +
97 std::abs(glm::dot(glm::vec3{ 0.f, 1.f, 0.f }, forward));
98
99 const float newIk = std::abs(glm::dot(glm::vec3{ 0.f, 0.f, 1.f }, right)) +
100 std::abs(glm::dot(glm::vec3{ 0.f, 0.f, 1.f }, up)) +
101 std::abs(glm::dot(glm::vec3{ 0.f, 0.f, 1.f }, forward));
102
103 const SquareAABB globalAABB(globalCenter, std::max(std::max(newIi, newIj), newIk));
104
105 return (globalAABB.isOnOrForwardPlane(camFrustum.leftFace) &&
106 globalAABB.isOnOrForwardPlane(camFrustum.rightFace) &&

127 std::array<glm::vec3, 8> getVertice() const

131 vertice[1] = { center.x + extents.x, center.y - extents.y, center.z - extents.z };
132 vertice[2] = { center.x - extents.x, center.y + extents.y, center.z - extents.z };
133 vertice[3] = { center.x + extents.x, center.y + extents.y, center.z - extents.z };

136 vertice[6] = { center.x - extents.x, center.y + extents.y, center.z + extents.z };
137 vertice[7] = { center.x + extents.x, center.y + extents.y, center.z + extents.z };
