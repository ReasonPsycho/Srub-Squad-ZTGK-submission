124inline void LevelGenerator::dfs(glm::ivec2 pos, int index, F&& f) noexcept(noexcept(f(glm::ivec2 {}))) {
125 auto tile = getTile(pos);
126 if (tile == nullptr || tile->pocketIndex != index || tile->dfsVisited)
127 return;
128 tile->dfsVisited = true;
129 f(pos);
130 for (const auto& dir : directions) {

142template<class Pred>
143inline std::vector<glm::ivec2> LevelGenerator::reservoirSamplePocket(int index, int count, PcgEngine& rand, Pred&& predicate) noexcept(noexcept(predicate(glm::ivec2 {}))) {
144 std::vector<glm::ivec2> result;
145 int soFar = 0;
146 forEachPocketTile(index, [&](glm::ivec2 pos) {
147 if (predicate(pos)) {
148 if (soFar < count) {
149 result.push_back(pos);
150 } else {
151 std::uniform_int_distribution<int> dist(0, soFar);
152 auto index = dist(rand);
153 if (index < count)
154 result[index] = pos;
155 }
156 soFar++;
157 }
158 });
159 return result;
160}

163inline int LevelGenerator::addAtRandomToPocket(int index, int count, LevelLayout::Tile::Type type, PcgEngine& rand, Pred&& predicate) noexcept(noexcept(predicate(glm::ivec2 {}))) {
164 auto positions = reservoirSamplePocket(index, count, rand, [&](glm::ivec2 pos) {
165 return getTile(pos)->type == Tile::Type::floor && predicate(pos);
166 });
167 for (const auto& pos : positions) {
168 getTile(pos)->type = type;
169 }
170 return static_cast<int>(positions.size());
